<script>
    let rowIdCounter = 0;
    const defaultColors = ["#f28b82", "#fcb92c", "#8ab4f8", "#81c995", "#c58af9", "#9aa0a6"];

    function toggleTheme(cb) {
        document.documentElement.setAttribute('data-theme', cb.checked ? 'light' : 'dark');
    }

    function togglePrediction(cb) {
        document.body.classList.toggle('prediction-mode', cb.checked);
    }

    function addRow() {
        rowIdCounter++;
        const list = document.getElementById('subject-list');
        const row = document.createElement('div');
        row.className = 'input-row';
        row.id = `input-row-${rowIdCounter}`;
        const colorValue = defaultColors[(rowIdCounter - 1) % defaultColors.length];
        
        row.innerHTML = `
            <span class="row-number">${rowIdCounter}</span>
            <input type="text" class="input-field name" placeholder="教科名">
            <input type="number" class="input-field score normal-only" placeholder="点数">
            <input type="number" class="input-field score-min prediction-only" placeholder="最小">
            <input type="number" class="input-field score-med prediction-only" placeholder="中央">
            <input type="number" class="input-field score-max prediction-only" placeholder="最大">
            <input type="number" class="input-field max" placeholder="満点" value="100">
            <input type="color" class="input-field color" value="${colorValue}">
            <button class="btn btn-delete" onclick="removeRow('${row.id}')">削除</button>
        `;
        list.appendChild(row);
    }

    function removeRow(id) {
        const row = document.getElementById(id);
        if (row) row.remove();
    }

    function createGraph() {
        const isPred = document.getElementById('prediction-checkbox').checked;
        const graphRowsContainer = document.getElementById('graph-rows');
        graphRowsContainer.innerHTML = '';
        document.getElementById('display-title').textContent = document.getElementById('exam-title').value || "模試結果";

        let total = { med: 0, min: 0, max: 0, full: 0 };
        let hasData = false;
        const rows = document.querySelectorAll('.input-row');

        rows.forEach(row => {
            const name = row.querySelector('.name').value.trim();
            const max = parseFloat(row.querySelector('.max').value) || 100;
            const color = row.querySelector('.color').value;
            
            let val = { med: 0, min: 0, max: 0 };
            let activeScore = 0;
            let isValid = false;

            if (isPred) {
                val.min = parseFloat(row.querySelector('.score-min').value) || 0;
                val.med = parseFloat(row.querySelector('.score-med').value) || 0;
                val.max = parseFloat(row.querySelector('.score-max').value) || 0;
                activeScore = val.med;
                if (row.querySelector('.score-med').value !== "") isValid = true;
            } else {
                activeScore = parseFloat(row.querySelector('.score').value) || 0;
                val.med = activeScore;
                if (row.querySelector('.score').value !== "") isValid = true;
            }

            if (isValid) {
                hasData = true;
                total.med += val.med;
                total.min += (isPred ? val.min : activeScore);
                total.max += (isPred ? val.max : activeScore);
                total.full += max;

                const getPer = (v) => Math.min((v / max) * 100, 100);
                const rowEl = document.createElement('div');
                rowEl.className = 'graph-row';
                
                let errorHtml = '';
                if (isPred) {
                    const minP = getPer(val.min);
                    const maxP = getPer(val.max);
                    errorHtml = `<div class="error-bar" style="left: ${minP}%; width: ${maxP - minP}%;"><div class="error-cap" style="left: 0;"></div><div class="error-cap" style="right: 0;"></div></div>`;
                }

                rowEl.innerHTML = `
                    <div class="row-labels">
                        <div class="label-color-bar" style="background-color: ${color};"></div>
                        <div class="label-name">${name || "教科"}</div>
                        <div class="label-score-box">
                            <span class="label-score">${activeScore}</span>
                            <span class="label-max">/${max}</span>
                        </div>
                    </div>
                    <div class="bar-area">
                        ${errorHtml}
                        <div class="bar" style="width: 0%; background-color: ${color};">${activeScore}</div>
                    </div>`;
                graphRowsContainer.appendChild(rowEl);
                setTimeout(() => rowEl.querySelector('.bar').style.width = `${getPer(activeScore)}%`, 100);
            }
        });

        if (hasData) {
            // 合計行の作成
            const getTotalPer = (v) => Math.min((v / total.full) * 100, 100);
            
            const divider = document.createElement('div');
            divider.className = 'total-divider';
            graphRowsContainer.appendChild(divider);

            const totalRow = document.createElement('div');
            totalRow.className = 'graph-row total-row';
            
            let totalErrorHtml = '';
            if (isPred) {
                const minP = getTotalPer(total.min);
                const maxP = getTotalPer(total.max);
                totalErrorHtml = `<div class="error-bar" style="left: ${minP}%; width: ${maxP - minP}%;"><div class="error-cap" style="left: 0;"></div><div class="error-cap" style="right: 0;"></div></div>`;
            }

            totalRow.innerHTML = `
                <div class="row-labels">
                    <div class="label-color-bar" style="background-color: var(--total-bar-color);"></div>
                    <div class="label-name" style="font-weight: bold;">合計</div>
                    <div class="label-score-box">
                        <span class="label-score">${total.med}</span>
                        <span class="label-max">/${total.full}</span>
                    </div>
                </div>
                <div class="bar-area">
                    ${totalErrorHtml}
                    <div class="bar" style="width: 0%; background-color: var(--total-bar-color);">${total.med}</div>
                </div>`;
            graphRowsContainer.appendChild(totalRow);
            setTimeout(() => totalRow.querySelector('.bar').style.width = `${getTotalPer(total.med)}%`, 150);

            document.getElementById('display').style.display = 'block';
            window.scrollTo({ top: document.getElementById('display').offsetTop, behavior: 'smooth' });
        }
    }

    // 初期行
    for(let i=0; i<3; i++) addRow();
</script>
